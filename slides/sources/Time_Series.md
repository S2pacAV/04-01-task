#  –°–∫–æ–ª—å–∑—è—â–µ–µ —Å—Ä–µ–¥–Ω–µ–µ

# –í–∑–≤–µ—à–µ–Ω–Ω–æ–µ —Å–∫–æ–ª—å–∑—è—â–µ–µ —Å—Ä–µ–¥–Ω–µ–µ

$$ \hat{y}_{t} = \displaystyle\sum^{k}_{n=1} \omega_n y_{t-n}$$

–í –∫–∞—á–µ—Å—Ç–≤–µ —Å–ª–µ–¥—É—é—â–µ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è –∏—Å–ø–æ–ª—å–∑—É–µ–º —Å—Ä–µ–¥–Ω–µ-–≤–∑–≤–µ—à–µ–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –ø–æ—Å–ª–µ–¥–Ω–∏—Ö $k$ –Ω–∞–±–ª—é–¥–µ–Ω–∏–π —Å –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–µ–º $\sum_{n=1}^{k}w_n=1$

--- 
# –≠–∫—Å–ø–æ–Ω–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–µ —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ

–í–∑–≤–µ—à–∏–≤–∞–µ–º __–≤—Å–µ__ –¥–æ—Å—Ç—É–ø–Ω—ã–µ –Ω–∞–±–ª—é–¥–µ–Ω–∏—è —Å —ç–∫—Å–ø–æ–Ω–µ–Ω—Ü–∏–∞–ª—å–Ω–æ —É–±—ã–≤–∞—é—â–∏–º–∏ –≤–µ—Å–∞–º–∏

$$\hat{y}_{t} = \alpha \cdot y_{t-1} + (1-\alpha) \cdot \hat y_{t-1}$$

* $y_t$ - –Ω–∞—Å—Ç–æ—è—â–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –≤—Ä–µ–º–µ–Ω–Ω–æ–≥–æ —Ä—è–¥–∞ –Ω–∞ —à–∞–≥–µ $t$
* $\alpha$ - –ø–∞—Ä–∞–º–µ—Ç—Ä —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏—è


---
# –î–≤–æ–π–Ω–æ–µ —ç–∫—Å–ø–æ–Ω–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–µ —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ

–†–∞–∑–±–∏–≤–∞–µ–º –Ω–∞—à–µ –∑–Ω–∞—á–µ–Ω–∏–µ –Ω–∞ –¥–≤–∞ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞: —É—Ä–æ–≤–µ–Ω—å –∏ —Ç—Ä–µ–Ω–¥

$$
\begin{aligned}
& \ell_t = \alpha y_t + (1-\alpha)(\ell_{t-1} + b_{t-1})\\ 
& b_t = \beta(\ell_t - \ell_{t-1}) + (1-\beta)b_{t-1}\\ 
& \hat{y}_{t+1} = \ell_t + b_t
\end{aligned}
$$

$b$ ‚Äì —Ç—Ä–µ–Ω–¥ 
$\ell$ ‚Äì —É—Ä–æ–≤–µ–Ω—å
$\hat{y}$ ‚Äì  –∏—Ç–æ–≥–æ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è –∫–∞–∫ —Å—É–º–º–∞ —Ç—Ä–µ–Ω–¥–∞ –∏ —É—Ä–æ–≤–Ω—è 

$\alpha$ ‚Äì –ø–∞—Ä–∞–º–µ—Ç—Ä —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏—è —Ä—è–¥–∞ –≤–æ–∫—Ä—É–≥ —Ç—Ä–µ–Ω–¥–∞
$\beta$ ‚Äì —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ —Ç—Ä–µ–Ω–¥–∞

--- 
# –¢—Ä–æ–π–Ω–æ–µ —ç–∫—Å–ø–æ–Ω–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–µ —Å–≥–ª–∞–∂–∏–≤–∞–Ω–∏–µ
__–ú–æ–¥–µ–ª—å –•–æ–ª—å—Ç–∞-–í–∏–Ω—Ç–µ—Ä—Å–∞__
</br>

$$
\begin{aligned}
&\ell_t = \alpha(y_t - s_{t-L}) + (1-\alpha)(\ell_{t-1} + b_{t-1})\\ 
&b_t = \beta(\ell_t - \ell_{t-1}) + (1-\beta)b_{t-1}\\ 
&s_t = \gamma(y_t - \ell_t) + (1-\gamma)s_{t-L}\\ 
&\hat{y}_{t+m} = l_t + mb_t + s_{t-L+1+(m-1)modL}
\end{aligned}
$$
